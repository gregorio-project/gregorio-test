% !TEX TS-program = lualatex+se

% This is a simple template for a LuaLaTeX document using gregorio scores.

\documentclass[11pt]{article} % use larger type; default would be 10pt

% usual packages loading:
\usepackage{fontspec}
\usepackage{graphicx} % support the \includegraphics command and options
\usepackage{geometry} % See geometry.pdf to learn the layout options. There are lots.
\geometry{a4paper} % or letterpaper (US) or a5paper or....
\usepackage[debug={knownline},allowdeprecated=false,autocompile]{gregoriotex} % for gregorio score inclusion
\usepackage{fullpage} % to reduce the margins
\setmainfont[
    Path = ../../../fonts/ ,
    Extension = .otf ,
    UprightFont = *-Regular ,
    UprightFeatures = { SmallCapsFont = *SC-Regular } ,
    BoldFont = *-Bold ,
    BoldFeatures = { SmallCapsFont = *SC-Bold } ,
    ItalicFont = *-Italic ,
    ItalicFeatures = { SmallCapsFont = *SC-Italic } ,
    BoldItalicFont = *-BoldItalic ,
    BoldItalicFeatures = { SmallCapsFont = *SC-BoldItalic } ,
    Ligatures = TeX
]{Alegreya}

% here we begin the document
\listfiles
\begin{document}

\gresetinitiallines{2}

2-line score
\gregorioscore{2-lines.gabc}

\newpage
3-line score
\gregorioscore{3-lines.gabc}

\newpage
4-line score
\gregorioscore{4-lines.gabc}

\newpage
5-line score
\gregorioscore{5-lines.gabc}

\end{document}
